    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const imageContainer = document.getElementById('notionImage');
            const likeCountElement = document.getElementById('likeCount');
            const notification = document.getElementById('notification');
            const closeBtn = document.getElementById('closeBtn');
            
            let likeCount = 0;
            
            // 从本地存储加载点赞数
            if (localStorage.getItem('likeCount')) {
                likeCount = parseInt(localStorage.getItem('likeCount'));
                likeCountElement.textContent = likeCount;
            }
            
            // 图片点击事件
            imageContainer.addEventListener('click', function(e) {
                // 防止点击关闭按钮时触发
                if (e.target !== closeBtn) {
                    createHeart(e.offsetX, e.offsetY);
                    updateLikeCount();
                    showNotification();
                }
            });
            
            // 关闭通知
            closeBtn.addEventListener('click', function() {
                imageContainer.querySelectorAll('.heart-emoji').forEach(heart => {
                    heart.remove();
                });
            });
            
            // 创建爱心元素
            function createHeart(x, y) {
                const heart = document.createElement('div');
                heart.classList.add('heart-emoji');
                heart.innerHTML = '❤️';
                heart.style.left = `${x}px`;
                heart.style.top = `${y}px`;
                
                imageContainer.appendChild(heart);
                
                // 动画结束后移除元素
                setTimeout(() => {
                    if (heart.parentNode) {
                        heart.remove();
                    }
                }, 1000);
            }
            
            // 更新点赞计数
            function updateLikeCount() {
                likeCount++;
                likeCountElement.textContent = likeCount;
                localStorage.setItem('likeCount', likeCount);
            }
            
            // 显示通知
            function showNotification() {
                notification.classList.remove('hidden');
                
                // 3秒后隐藏通知
                setTimeout(() => {
                    notification.classList.add('hidden');
                }, 3000);
            }
        });
    </script>
